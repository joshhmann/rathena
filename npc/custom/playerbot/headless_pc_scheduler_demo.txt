// Dev-only world scheduler proof for shared headless social controllers.

prontera,148,183,4	script	Headless Scheduler	4_M_MANAGER,{
	.controller$ = "HeadlessWorldScheduler";
	mes "[Headless Scheduler]";
	switch(select("Start scheduler:Stop scheduler:Status:Controller drill-down:Toggle controller enabled:Reload control plane:Cancel")) {
	case 1:
		callfunc "F_PB_DB_LoadAllPools";
		callfunc "F_LW_HPC_SchedStart", .controller$;
		mes "World scheduler enabled.";
		close;
	case 2:
		set getvariableofnpc(.enabled, .controller$), 0;
		donpcevent .controller$ + "::OnStop";
		mes "World scheduler disabled.";
		close;
	case 3:
		callfunc "F_LW_HPC_SchedBuildStatus", .controller$, getvariableofnpc(.enabled, .controller$);
		mes getvariableofnpc(.status$, .controller$);
		close;
	case 4:
		.@choice = select(callfunc("F_PB_DB_ControllerMenu$"));
		if (.@choice > $@pbctl_count)
			close;
		mes callfunc("F_PB_DB_BuildControllerSummary$", .@choice - 1);
		close;
	case 5:
		.@choice = select(callfunc("F_PB_DB_ControllerMenu$"));
		if (.@choice > $@pbctl_count)
			close;
		.@index = .@choice - 1;
		callfunc "F_PB_DB_SetControllerEnabled", $@pbctl_key$[.@index], !$@pbctl_enabled[.@index];
		callfunc "F_PB_DB_ReloadControlPlane", .controller$;
		mes "Controller " + $@pbctl_label$[.@index] + " is now " + (!$@pbctl_enabled[.@index] ? "enabled" : "disabled") + ".";
		close;
	case 6:
		callfunc "F_PB_DB_ReloadControlPlane", .controller$;
		mes "Reloaded playerbot control plane.";
		mes "Pools: " + $@pbctl_pools + " loaded. Controllers: " + $@pbctl_ctrls + " re-primed. Merchant reconciles: " + $@pbctl_merchants + ".";
		close;
	default:
		close;
	}
}

-	script	HeadlessWorldScheduler	FAKE_NPC,{
OnInit:
	.enabled = 0;
	.running = 0;
	.tick_pending = 0;
	callsub S_Prime;
	end;

OnReload:
	.tick_pending = 0;
	if (playerattached() > 0)
		deltimer strnpcinfo(3) + "::OnTick";
	callsub S_Prime;
	if (.enabled)
		callfunc "F_LW_HPC_SchedQueueTick", strnpcinfo(3), 0;
	end;

OnTick:
	if (.running)
		end;
	.tick_pending = 0;
	.running = 1;
	if (!.enabled)
		goto L_Done;
	callfunc "F_LW_HPC_SchedRun", strnpcinfo(3);
	callfunc "F_LW_HPC_SchedNext", strnpcinfo(3);
L_Done:
	.running = 0;
	end;

OnStop:
	.enabled = 0;
	.tick_pending = 0;
	if (playerattached() > 0)
		deltimer strnpcinfo(3) + "::OnTick";
	callfunc "F_PB_DB_StopControllers";
	return;

S_Prime:
	callfunc "F_LW_HPC_SchedReset", strnpcinfo(3);
	callfunc "F_LW_HPC_SchedSetTickMs", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sw.tm", 5000);
	callfunc "F_LW_HPC_SchedSetCap", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sw.cc", 2);
	callfunc "F_LW_HPC_SchedSetActorCap", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sw.ac", 5);
	callfunc "F_LW_HPC_SchedSetMapCap", strnpcinfo(3), "prontera", callfunc("F_PB_CFG_GetInt", "sw.mp.ac", 5);
	callfunc "F_LW_HPC_SchedSetMapCap", strnpcinfo(3), "alberta", callfunc("F_PB_CFG_GetInt", "sw.ma.ac", 5);
	callfunc "F_PB_DB_LoadSchedulerControllers", strnpcinfo(3);
	return;
}

-	script	PlayerbotSchedulerSelftest	FAKE_NPC,{
OnInterIfInitOnce:
	.enabled = 0;
	if (!.enabled)
		end;
	.base_ok = 0;
	.disable_ok = 0;
	.enable_ok = 0;
	.@count = callfunc("F_PB_DB_LoadControllerRegistry");
	.base_ok = (.@count >= 3);
	initnpctimer;
	end;

OnTimer100:
	callfunc "F_PB_DB_SetControllerEnabled", "merchant.alberta", 0;
	callfunc "F_PB_DB_ReloadControlPlane", "HeadlessWorldScheduler";
	.@after_disable = callfunc("F_PB_DB_LoadControllerRegistry");
	.disable_ok = (.@after_disable >= 3);
	for (.@i = 0; .@i < $@pbctl_count; ++.@i) {
		if ($@pbctl_key$[.@i] == "merchant.alberta")
			.disable_ok = (.disable_ok && !$@pbctl_enabled[.@i]);
	}
	end;

OnTimer200:
	callfunc "F_PB_DB_SetControllerEnabled", "merchant.alberta", 1;
	callfunc "F_PB_DB_ReloadControlPlane", "HeadlessWorldScheduler";
	.@after_enable = callfunc("F_PB_DB_LoadControllerRegistry");
	.enable_ok = (.@after_enable >= 3);
	for (.@i = 0; .@i < $@pbctl_count; ++.@i) {
		if ($@pbctl_key$[.@i] == "merchant.alberta")
			.enable_ok = (.enable_ok && $@pbctl_enabled[.@i]);
	}
	debugmes "playerbot_scheduler_selftest: base_ok=" + .base_ok + " disable_ok=" + .disable_ok + " enable_ok=" + .enable_ok + " result=" + (.base_ok && .disable_ok && .enable_ok) + ".";
	stopnpctimer;
	end;
}
