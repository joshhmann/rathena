// Dev-only world scheduler proof for shared headless social controllers.

prontera,148,183,4	script	Headless Scheduler	4_M_MANAGER,{
	.controller$ = "HeadlessWorldScheduler";
	mes "[Headless Scheduler]";
	switch(select("Start scheduler:Stop scheduler:Status:Controller drill-down:Cancel")) {
	case 1:
		set getvariableofnpc(.enabled, .controller$), 1;
		donpcevent .controller$ + "::OnTick";
		mes "World scheduler enabled.";
		close;
	case 2:
		set getvariableofnpc(.enabled, .controller$), 0;
		donpcevent .controller$ + "::OnStop";
		mes "World scheduler disabled.";
		close;
	case 3:
		callfunc "F_LW_HPC_SchedBuildStatus", .controller$, getvariableofnpc(.enabled, .controller$);
		mes getvariableofnpc(.status$, .controller$);
		close;
	case 4:
		switch(select("Prontera social:Alberta social:Cancel")) {
		case 1:
			.@target$ = "HeadlessPronteraSocialController";
			break;
		case 2:
			.@target$ = "HeadlessAlbertaSocialController";
			break;
		default:
			close;
		}
		callfunc "F_LW_HPC_DefBuildStatus", .@target$, getvariableofnpc(.enabled, .@target$);
		mes "[" + .@target$ + "]";
		mes getvariableofnpc(.status$, .@target$);
		close;
	default:
		close;
	}
}

-	script	HeadlessWorldScheduler	FAKE_NPC,{
OnInit:
	.enabled = 0;
	callfunc "F_LW_HPC_SchedReset", strnpcinfo(3);
	callfunc "F_LW_HPC_SchedSetTickMs", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sched.world.tick_ms", 5000);
	callfunc "F_LW_HPC_SchedSetCap", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sched.world.controller_cap", 2);
	callfunc "F_LW_HPC_SchedSetActorCap", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sched.world.actor_cap", 5);
	callfunc "F_LW_HPC_SchedSetMapCap", strnpcinfo(3), "prontera", callfunc("F_PB_CFG_GetInt", "sched.world.map.prontera.actor_cap", 5);
	callfunc "F_LW_HPC_SchedSetMapCap", strnpcinfo(3), "alberta", callfunc("F_PB_CFG_GetInt", "sched.world.map.alberta.actor_cap", 5);
	callfunc "F_LW_HPC_SchedAdd", strnpcinfo(3), "HeadlessPronteraSocialController", "prontera",
		callfunc("F_PB_CFG_GetInt", "sched.world.ctrl.prontera.gate_users", 1),
		callfunc("F_PB_CFG_GetInt", "sched.world.ctrl.prontera.priority", 90),
		callfunc("F_PB_CFG_GetInt", "sched.world.ctrl.prontera.actor_weight", 5);
	callfunc "F_LW_HPC_SchedAdd", strnpcinfo(3), "HeadlessAlbertaSocialController", "alberta",
		callfunc("F_PB_CFG_GetInt", "sched.world.ctrl.alberta.gate_users", 1),
		callfunc("F_PB_CFG_GetInt", "sched.world.ctrl.alberta.priority", 80),
		callfunc("F_PB_CFG_GetInt", "sched.world.ctrl.alberta.actor_weight", 5);
	end;

OnTick:
	if (!.enabled)
		end;
	callfunc "F_LW_HPC_SchedRun", strnpcinfo(3);
	callfunc "F_LW_HPC_SchedNext", strnpcinfo(3);
	end;

OnStop:
	donpcevent "HeadlessPronteraSocialController::OnStop";
	donpcevent "HeadlessAlbertaSocialController::OnStop";
	return;
}
