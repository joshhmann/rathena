// Dev-only world scheduler proof for shared headless social controllers.

prontera,148,183,4	script	Headless Scheduler	4_M_MANAGER,{
	.controller$ = "HeadlessWorldScheduler";
	mes "[Headless Scheduler]";
	switch(select("Start scheduler:Stop scheduler:Status:Controller drill-down:Reload control plane:Cancel")) {
	case 1:
		callfunc "F_PB_DB_LoadAllPools";
		callfunc "F_LW_HPC_SchedStart", .controller$;
		mes "World scheduler enabled.";
		close;
	case 2:
		set getvariableofnpc(.enabled, .controller$), 0;
		donpcevent .controller$ + "::OnStop";
		mes "World scheduler disabled.";
		close;
	case 3:
		callfunc "F_LW_HPC_SchedBuildStatus", .controller$, getvariableofnpc(.enabled, .controller$);
		mes getvariableofnpc(.status$, .controller$);
		close;
	case 4:
		switch(select("Prontera social:Alberta social:Alberta merchant:Cancel")) {
		case 1:
			.@target$ = "HeadlessPronteraSocialController";
			break;
		case 2:
			.@target$ = "HeadlessAlbertaSocialController";
			break;
		case 3:
			.@target$ = "HeadlessAlbertaMerchantController";
			break;
		default:
			close;
		}
		callfunc "F_LW_HPC_DefBuildStatus", .@target$, getvariableofnpc(.enabled, .@target$);
		mes "[" + .@target$ + "]";
		mes getvariableofnpc(.status$, .@target$);
		close;
	case 5:
		callfunc "F_PB_DB_ReloadControlPlane", .controller$;
		mes "Reloaded playerbot control plane.";
		mes "Pools: " + $@pbctl_pools + " loaded. Controllers: " + $@pbctl_ctrls + " re-primed.";
		close;
	default:
		close;
	}
}

-	script	HeadlessWorldScheduler	FAKE_NPC,{
OnInit:
	.enabled = 0;
	.running = 0;
	.tick_pending = 0;
	callsub S_Prime;
	end;

OnReload:
	.tick_pending = 0;
	deltimer strnpcinfo(3) + "::OnTick";
	callsub S_Prime;
	if (.enabled)
		callfunc "F_LW_HPC_SchedQueueTick", strnpcinfo(3), 0;
	end;

OnTick:
	if (.running)
		end;
	.tick_pending = 0;
	.running = 1;
	if (!.enabled)
		goto L_Done;
	callfunc "F_LW_HPC_SchedRun", strnpcinfo(3);
	callfunc "F_LW_HPC_SchedNext", strnpcinfo(3);
L_Done:
	.running = 0;
	end;

OnStop:
	.enabled = 0;
	.tick_pending = 0;
	deltimer strnpcinfo(3) + "::OnTick";
	donpcevent "HeadlessPronteraSocialController::OnStop";
	donpcevent "HeadlessAlbertaSocialController::OnStop";
	donpcevent "HeadlessAlbertaMerchantController::OnStop";
	return;

S_Prime:
	callfunc "F_LW_HPC_SchedReset", strnpcinfo(3);
	callfunc "F_LW_HPC_SchedSetTickMs", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sw.tm", 5000);
	callfunc "F_LW_HPC_SchedSetCap", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sw.cc", 2);
	callfunc "F_LW_HPC_SchedSetActorCap", strnpcinfo(3), callfunc("F_PB_CFG_GetInt", "sw.ac", 5);
	callfunc "F_LW_HPC_SchedSetMapCap", strnpcinfo(3), "prontera", callfunc("F_PB_CFG_GetInt", "sw.mp.ac", 5);
	callfunc "F_LW_HPC_SchedSetMapCap", strnpcinfo(3), "alberta", callfunc("F_PB_CFG_GetInt", "sw.ma.ac", 5);
	callfunc "F_PB_DB_LoadSchedulerControllers", strnpcinfo(3);
	return;
}
