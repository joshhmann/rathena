// Dev-only Alberta merchant/social controller proof using the shared headless controller kit.

alberta,48,238,4	script	Headless Alberta Social	4_M_MANAGER,{
	.controller$ = "HeadlessAlbertaSocialController";
	mes "[Headless Alberta Social]";
	switch(select("Start market traffic:Stop market traffic:Status:Cancel")) {
	case 1:
		callfunc "F_LW_HPC_ControllerStart", .controller$;
		mes "Alberta social controller enabled.";
		close;
	case 2:
		donpcevent .controller$ + "::OnStop";
		mes "Alberta social controller disabled.";
		close;
	case 3:
		callfunc "F_LW_HPC_DefBuildStatus", .controller$, getvariableofnpc(.enabled, .controller$);
		mes getd("$LWHS" + getnpcid(0, .controller$) + "$");
		close;
	default:
		close;
	}
}

-	script	HeadlessAlbertaSocialController	FAKE_NPC,{
OnInit:
	.owner$ = strnpcinfo(3);
	.enabled = 0;
	callfunc "F_LW_HPC_DefReset", .owner$;
	callfunc "F_LW_HPC_DefSetTickMs", .owner$, callfunc("F_PB_CFG_GetInt", "ctrl.alberta.tick_ms", 2400);
	callfunc "F_LW_HPC_DefSetMapGate", .owner$, "alberta", callfunc("F_PB_CFG_GetInt", "ctrl.alberta.gate_users", 1);
	callfunc "F_LW_HPC_DefSetStartStagger", .owner$,
		callfunc("F_PB_CFG_GetInt", "ctrl.alberta.start_min_ms", 500),
		callfunc("F_PB_CFG_GetInt", "ctrl.alberta.start_max_ms", 1800);
	callfunc "F_LW_HPC_DefSetGraceMs", .owner$, callfunc("F_PB_CFG_GetInt", "ctrl.alberta.grace_ms", 15000);
	callfunc "F_LW_HPC_DefSetStopPolicy", .owner$, callfunc("F_PB_CFG_GetStr", "ctrl.alberta.stop_policy", "park");
	callfunc "F_LW_HPC_DefSetRoutineGroup", .owner$, callfunc("F_PB_CFG_GetStr", "ctrl.alberta.routine_group", "night");
	callfunc "F_LW_HPC_DefSetRoutineWindow", .owner$,
		callfunc("F_PB_CFG_GetInt", "ctrl.alberta.routine_start", 0),
		callfunc("F_PB_CFG_GetInt", "ctrl.alberta.routine_end", 6);
	callfunc "F_PB_CFG_ApplyControllerDef", .owner$, "social.alberta";
	end;

OnTick:
	if (!.enabled)
		end;

	if (!callfunc("F_LW_HPC_ControllerShouldRun", .owner$)) {
		callfunc "F_LW_HPC_DefStop", .owner$, .owner$;
		callfunc "F_LW_HPC_ControllerScheduleNext", strnpcinfo(3);
		end;
	}

	for (.@i = 0; .@i < callfunc("F_LW_HPC_DefActorCount", .owner$); ++.@i)
		callfunc "F_LW_HPC_DefTickActor", .owner$, .owner$, .@i;

	callfunc "F_LW_HPC_ControllerScheduleNext", strnpcinfo(3);
	end;

OnStop:
	.enabled = 0;
	callfunc "F_LW_HPC_DefStop", .owner$, .owner$;
	return;
}
