// Headless BL_PC Phase 0 lab harness.

prontera,154,181,4	script	Headless PC Lab	4_F_02,{
	mes "[Headless PC Lab]";
	mes "This is a dev-only harness for Phase 0 inert headless PCs.";
	next;
	switch(select("Spawn by char_id:Status by char_id:Spawn Ack by char_id:Remove Ack by char_id:Remove by char_id:Cancel")) {
	case 1:
		input .@char_id;
		input .@map$;
		input .@x;
		input .@y;
		if (headlesspc_spawn(.@char_id, .@map$, .@x, .@y))
			mes "Spawn request queued.";
		else
			mes "Spawn request failed.";
		close;
	case 2:
		input .@char_id;
		.@status = headlesspc_status(.@char_id);
		switch(.@status) {
		case 1:
			mes "Status: pending spawn.";
			break;
		case 2:
			mes "Status: active headless PC.";
			break;
		case 3:
			mes "Status: pending remove/save.";
			break;
		case 4:
			mes "Status: occupied by a non-headless live player.";
			break;
		default:
			mes "Status: absent.";
			break;
		}
		close;
	case 3:
		input .@char_id;
		mes "Spawn-ready ack seq: " + headlesspc_spawnack(.@char_id) + ".";
		close;
	case 4:
		input .@char_id;
		mes "Remove/save ack seq: " + headlesspc_ack(.@char_id) + ".";
		close;
	case 5:
		input .@char_id;
		if (headlesspc_remove(.@char_id))
			mes "Headless PC removed or pending spawn cleared.";
		else
			mes "Nothing to remove.";
		close;
	default:
		close;
	}
}
