// Temporary dev-only smoke harness for fixed headless PC spawn/remove.

prontera,154,179,4	script	Headless Smoke	4_F_01,{
	mes "[Headless Smoke]";
	switch(select("Spawn codexalt:Remove codexalt:Cancel")) {
	case 1:
		if (headlesspc_spawn(150002, "prontera", 160, 186))
			mes "Spawn request queued for codexalt.";
		else
			mes "Spawn request failed.";
		close;
	case 2:
		if (headlesspc_remove(150002))
			mes "Remove request queued for codexalt.";
		else
			mes "Nothing to remove.";
		close;
	default:
		close;
	}
}
